import re

t = int(input())

for _ in range(t):
    pattern = input()
    
    # Check for invalid consecutive repetition symbols
    if re.search(r'[\*\+\?]{2,}', pattern):
        print(False)
        continue
    
    try:
        re.compile(pattern)
        print(True)
    except re.error:
        print(False)
