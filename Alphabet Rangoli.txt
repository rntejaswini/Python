def print_rangoli(size):
    import string
    alpha = string.ascii_lowercase
    width = 4 * size - 3

    rows = []

    for i in range(size):
        s = "-".join(alpha[size-1:i:-1] + alpha[i:size])
        rows.append(s.center(width, "-"))

    # Print full rangoli
    print("\n".join(rows[::-1] + rows[1:]))
if __name__ == '__main__':
    n = int(input())
    print_rangoli(n)
